Express 集成 Swagger

准备

express-jsdoc-swagger

$ npm i express-jsdoc-swagger

集成步骤

  1. 创建 swaggerOption.js文件

    const options = {
  info: {
    version: '1.0.0',
    title: 'Albums store',
    description:
      "description",
    termsOfService: "http:///terms/",
    contact: { "name": "Support", "email": "" },
    license: {
      name: 'MIT',
      url: "http://www.apache.org/licenses/LICENSE-2.0.html",
    },
  },
  // 设置basicAuth
  security: {
    BasicAuth: {
      type: 'http',
      scheme: 'basic',
    },
  },
  // 设置bearerAuth
  //security: {
  //   bearerAuth: {
  //     type: 'http', 
  //     scheme: "bearer"
  //   },
  //},
  // Base directory which we use to locate your JSDOC files
  baseDir: __dirname,
  // Glob pattern to find your jsdoc files (multiple patterns can be added in an array)
  filesPattern: './**/*.js',
  // URL where SwaggerUI will be rendered
  swaggerUIPath: '/api-docs',
  // Expose OpenAPI UI
  exposeSwaggerUI: true,
  // Expose Open API JSON Docs documentation in `apiDocsPath` path.
  exposeApiDocs: false,
  // Open API JSON Docs endpoint.
  apiDocsPath: '/v3/api-docs',
  // Set non-required fields as nullable by default
  notRequiredAsNullable: false,
  // You can customize your UI options.
  // you can extend swagger-ui-express config. You can checkout an example of this
  // in the `example/configuration/swaggerOptions.js`
  swaggerUiOptions: {},
  // multiple option in case you want more that one instance
  multiple: true,
};
module.exports = options;

  2. 在app.js中引入

    var app = express();

expressJSDocSwagger(app)(require('./swaggerOption.js'));

  3. api 注释使用

    /**
 * QueryProductList
 * @typedef {object} QueryProductList
 * @property {string} accountId - accountId
 * @property {string} sort      - list by sort
 * @property {string} page.required - page number
 * @property {string} limit.required -number per page
 */
/**
 * @typedef {object}ProductListResponse
 * @property {Array.<InsertProduct>} products - 产品列表数据
 * @property {number} totalCount - 总计数
 * @property {number} page - 当前页码
 * @property {number} pageSize - 每页大小
 */
/**
 * post /product/lists
 * @tags Product
 * @description get products list
 * @summary products list
 * @security bearerAuth
 * @param {QueryProductList} request.body.required - query info - application/json
 * @return {ProductListResponse} 200 - successful operation
 * @return {object} 400 - Bad Request
 * @example request - example JSON format
 * {
 * "accountId": "",
 * "sort": {
 *    "cdate":-1
 * },
 * "page":1,
 * "limit":1
 * }
 * @example response - 400 - example JSON format
 * {
 * "statusCode": 400,
 * "msg": "Error Message",
 * "result":null
 * }
 * @example response - 200 - example JSON format
 * {
 * "statusCode": 200,
 * "msg": "",
 * "result":{ 
 *    "products":[]
 * }
 * }
 */
app.get('/api/v1', (req, res) => res.json({
  success: true,
}));
